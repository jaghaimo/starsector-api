66 Global.getSoundPlayer().setSuspendDefaultMusicPlayback(true);

68 } else if (SOE_STATE_POST_DUEL.equals(state)) {
69 Global.getSoundPlayer().setSuspendDefaultMusicPlayback(true);
70 Global.getSoundPlayer().pauseCustomMusic();

73 }
74
75 if (SOE_STATE_PARTY.equals(state)) {
76 Global.getSoundPlayer().playUILoop(PARTY_CHATTER, 1f, 1f);
77 } else if (SOE_STATE_PRE_DUEL.equals(state)) {
78 Global.getSoundPlayer().playUILoop(PARTY_CHATTER, 1f, 0.5f);
79 }
80
81 if (!Global.getSector().isPaused()) {
82 stop();
83 }
