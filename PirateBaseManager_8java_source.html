193 if (days < 45f) continue;

196 float weight = 0f;
197 if (system.hasTag(Tags.THEME_MISC_SKIP)) {

206 weight = 5f;

208 //weight = 1f;
209 weight = 0f;

211 if (weight <= 0f) continue;
212
213 float usefulStuff = system.getCustomEntitiesWithTag(Tags.OBJECTIVE).size() +
214 system.getCustomEntitiesWithTag(Tags.STABLE_LOCATION).size();
215 if (usefulStuff <= 0) continue;
216
217 if (Misc.hasPulsar(system)) continue;
218 if (Misc.getMarketsInLocation(system).size() > 0) continue;
219

221
