95 ShipVariantAPI variant = null;

97 List<ShipRolePick> picks = market.getFaction().pickShip(role, params, null, genRandom);

110 member = Global.getFactory().createFleetMember(FleetMemberType.SHIP, variant);
111 assignShipName(member, Factions.INDEPENDENT);
112
113 float quality = ShipQuality.getShipQuality(market, person.getFaction().getId());
114 float averageDmods = DefaultFleetInflater.getAverageDmodsForQuality(quality);
115 int addDmods = DefaultFleetInflater.getNumDModsToAdd(variant, averageDmods, genRandom);
116 if (addDmods > 0) {
117 DModManager.setDHull(variant);
118 DModManager.addDMods(member, true, addDmods, genRandom);
119 }
120 member.getCrewComposition().setCrew(100000);
121 member.getRepairTracker().setCR(0.7f);
122
123 if (BASE_PRICE_MULT == 1f) {
124 price = (int) Math.round(variant.getHullSpec().getBaseValue());
125 } else {
126 price = getRoundNumber(variant.getHullSpec().getBaseValue() * BASE_PRICE_MULT);
127 }
128
129 setRepFactionChangesTiny();
130 setRepPersonChangesVeryLow();
131
132 return true;
133 }

179 public static String pickRole(float quality, FactionAPI faction, PersonImportance imp, Random random) {
180 WeightedRandomPicker<String> picker = new WeightedRandomPicker<String>(random);
181
182 float cycles = PirateBaseManager.getInstance().getDaysSinceStart() / 365f;
183
184 if (imp == PersonImportance.VERY_HIGH && cycles < 3) imp = PersonImportance.HIGH;
185 if (imp == PersonImportance.HIGH && cycles < 1) imp = PersonImportance.MEDIUM;
186
187 if (quality < 0.5f && imp.ordinal() > PersonImportance.MEDIUM.ordinal()) {
188 imp = PersonImportance.MEDIUM;
189 }
190
191 float w = faction.getDoctrine().getWarships() - 1f;
192 float c = faction.getDoctrine().getCarriers() - 1f;
193 float p = faction.getDoctrine().getPhaseShips() - 1f;
194 if (w + c + p < 1) w = 1;
195
196 switch (imp) {
197 case VERY_LOW:
198 picker.add(ShipRoles.COMBAT_SMALL, w);
199 picker.add(ShipRoles.COMBAT_MEDIUM, w/2f);
200 picker.add(ShipRoles.CARRIER_SMALL, c);
201 picker.add(ShipRoles.PHASE_SMALL, p);
202 break;
203 case LOW:
204 picker.add(ShipRoles.COMBAT_SMALL, w/2f);
205 picker.add(ShipRoles.COMBAT_MEDIUM, w);
206 picker.add(ShipRoles.CARRIER_SMALL, c);
207 picker.add(ShipRoles.PHASE_SMALL, p);
208 break;
209 case MEDIUM:
210 picker.add(ShipRoles.COMBAT_MEDIUM, w/2f);
211 picker.add(ShipRoles.COMBAT_LARGE, w);
212 picker.add(ShipRoles.CARRIER_SMALL, c/2f);
213 picker.add(ShipRoles.CARRIER_MEDIUM, c);
214 picker.add(ShipRoles.PHASE_SMALL, p/2f);
215 picker.add(ShipRoles.PHASE_MEDIUM, p);
216 break;
217 case HIGH:
218 picker.add(ShipRoles.COMBAT_MEDIUM, w/4f);
219 picker.add(ShipRoles.COMBAT_LARGE, w);
220 picker.add(ShipRoles.COMBAT_CAPITAL, w/2f);
221 picker.add(ShipRoles.CARRIER_MEDIUM, c);
222 picker.add(ShipRoles.CARRIER_LARGE, c/2f);
223 picker.add(ShipRoles.PHASE_MEDIUM, p);
224 picker.add(ShipRoles.PHASE_LARGE, p/2f);
225 break;
226 case VERY_HIGH:
227 picker.add(ShipRoles.COMBAT_MEDIUM, w/4f);
228 picker.add(ShipRoles.COMBAT_LARGE, w/2f);
229 picker.add(ShipRoles.COMBAT_CAPITAL, w);
230 picker.add(ShipRoles.CARRIER_MEDIUM, c/2f);
231 picker.add(ShipRoles.CARRIER_LARGE, c);
232 picker.add(ShipRoles.PHASE_MEDIUM, p/2f);
233 picker.add(ShipRoles.PHASE_LARGE, p);
234 picker.add(ShipRoles.PHASE_CAPITAL, p/2f);
235 break;
236 }
237 return picker.pick();
238 }
