133 protected void updateSpawnRateMult() {
134 if (currSpawnLoc == null) {
135 if (DEBUG) {
136 System.out.println("No target system, spawnRateMult is 1");
137 }
138 spawnRateMult = 1f;
139 return;
140 }
141
142 float desiredNumFleets = getDesiredNumFleetsForSpawnLocation();
143 float recentSpawns = getRecentSpawnsForSystem(currSpawnLoc);
144 if (active != null) {
145 float activeInSystem = 0f;
146 for (ManagedFleetData data : active) {
147 if (data.spawnedFor == currSpawnLoc || data.fleet.getContainingLocation() == currSpawnLoc) {
148 activeInSystem++;
149 }
150 }
151 recentSpawns = Math.max(recentSpawns, activeInSystem);
152 }
153
154 spawnRateMult = (float) Math.pow(Math.max(0, (desiredNumFleets - recentSpawns) * 1f), 4f);
155 if (spawnRateMult < 0) spawnRateMult = 0;

176 protected StarSystemAPI pickNearestPopulatedSystem() {
177 if (Global.getSector().isInNewGameAdvance()) return null;
178 CampaignFleetAPI player = Global.getSector().getPlayerFleet();
179 if (player == null) return null;
180 StarSystemAPI nearest = null;
181 float minDist = Float.MAX_VALUE;
182 for (MarketAPI market : Global.getSector().getEconomy().getMarketsCopy()) {
183 if (market.isHidden()) continue;
184
185 if (market.isPlayerOwned() && market.getSize() <= 3) continue;
186 if (!market.hasSpaceport()) continue;
187
188 float distToPlayerLY = Misc.getDistanceLY(player.getLocationInHyperspace(), market.getLocationInHyperspace());
189
190 if (distToPlayerLY > MAX_RANGE_FROM_PLAYER_LY) continue;
191
192 if (distToPlayerLY < minDist && market.getStarSystem() != null) {
193 if (market.getStarSystem().getStar() != null) {
194 if (market.getStarSystem().getStar().getSpec().isPulsar()) continue;
195 }
196
197 nearest = market.getStarSystem();
198 minDist = distToPlayerLY;
199 }
200 }
201
202
203 // stick with current system longer unless something else is closer
204 if (nearest == null && currSpawnLoc != null) {
205 float distToPlayerLY = Misc.getDistanceLY(player.getLocationInHyperspace(), currSpawnLoc.getLocation());
206 if (distToPlayerLY <= DESPAWN_RANGE_LY) {
207 nearest = currSpawnLoc;
208 }
209 }
210
211 return nearest;
212 }

214 public CampaignFleetAPI spawnFleet() {
215 if (currSpawnLoc == null) return null;
216
217 // otherwise, possible for jump-point dialog to say there's nothing on other side
218 // but there will be by the time the player comes out
219 if (Global.getSector().getPlayerFleet() != null && Global.getSector().getPlayerFleet().isInHyperspaceTransition()) {
220 return null;
221 }
222
223 CampaignFleetAPI fleet = spawnFleetImpl();
224 if (fleet != null) {
225 fleet.getMemoryWithoutUpdate().set(KEY_SYSTEM, currSpawnLoc.getName());
226 fleet.getMemoryWithoutUpdate().set(KEY_SPAWN_FP, fleet.getFleetPoints());
227 }
228
229 // do this even if fleet is null, to avoid non-stop fail-spawning of fleets
230 // if spawnFleetImpl() can't spawn one, for whatever reason
231 addRecentSpawn(currSpawnLoc);
232 updateSpawnRateMult();
233
234 return fleet;
235 }
