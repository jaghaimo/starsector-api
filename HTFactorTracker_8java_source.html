207 protected void checkSystemCenter() {

209 StarSystemAPI system = pf.getStarSystem();

212 String type = null;

214 switch (system.getType()) {

219 case NEBULA:
220 type = "Center of starless nebula";

222 break;
223 case TRINARY_1CLOSE_1FAR:
224 type = "Center of binary system";
225 points = HTPoints.SCAN_BINARY;
226 break;
227 case TRINARY_2CLOSE:
228 type = "Center of trinary system";
229 points = HTPoints.SCAN_TRINARY;
230 break;
231 default:
232 int count = 0;
233 for (PlanetAPI curr : system.getPlanets()) {
234 if (!curr.isStar()) continue;
235 float dist = Misc.getDistance(curr.getLocation(), pf.getLocation());
236 if (dist < 2000 + curr.getRadius()) {
237 count++;
238 }
239 }
240 if (count > 1) {
241 type = "Stellar conflux";
242 if (count == 2) {
243 points = HTPoints.SCAN_BINARY;
244 } else {
245 points = HTPoints.SCAN_TRINARY;
246 }
247 }
248 break;
249 }
250 if (type == null) return;
251
252 String id = "systemtypescan_" + system.getId();
253 float range = pf.getLocation().length();
254
255 if (range < 2000) {
256 if (scanned.contains(id)) {
257 addMessage("Center of star system already scanned");
258 } else {
259 HyperspaceTopographyEventIntel.addFactorCreateIfNecessary(
260 new HTScanFactor(type + " scanned (" + system.getBaseName() + ")", points), null);
