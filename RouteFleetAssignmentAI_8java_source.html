47 protected TravelState getTravelState(RouteSegment segment) {

143 protected void pickNext(boolean justSpawned) {
144 RouteSegment current = route.getCurrent();
145 if (current == null) return;
146
147 List<RouteSegment> segments = route.getSegments();
148 int index = route.getSegments().indexOf(route.getCurrent());
149
150
151 if (index == 0 && route.getMarket() != null && !current.isTravel()) {
152 if (current.getFrom() != null && (current.getFrom().isSystemCenter() || current.getFrom().getMarket() != route.getMarket())) {
153 addLocalAssignment(current, justSpawned);
154 } else {
155 addStartingAssignment(current, justSpawned);
156 }
157 return;
158 }
159
160 if (index == segments.size() - 1 && route.getMarket() != null && !current.isTravel()
161 && (current.elapsed >= current.daysMax || current.getFrom() == route.getMarket().getPrimaryEntity())) {
162 addEndingAssignment(current, justSpawned);
163 return;
164 }
165
166 // transiting from current to next; may or may not be in the same star system
167 if (current.isTravel()) {
168 if (index == segments.size() - 1 &&
169 fleet.getContainingLocation() == current.to.getContainingLocation() &&
170 current.elapsed >= current.daysMax) {
171 addEndingAssignment(current, justSpawned);
172 } else {
173 addTravelAssignment(current, justSpawned);
174 }
175 return;
176 }
177
178 // in a system or in a hyperspace location for some time
179 if (!current.isTravel()) {
180 addLocalAssignment(current, justSpawned);
181 }
182 }
