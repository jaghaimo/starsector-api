138 protected ShipAPI findTarget(ShipAPI ship) {
139 float range = getMaxRange(ship);
140 boolean player = ship == Global.getCombatEngine().getPlayerShip();
141 ShipAPI target = ship.getShipTarget();
142
143 if (ship.getShipAI() != null && ship.getAIFlags().hasFlag(AIFlags.TARGET_FOR_SHIP_SYSTEM)){
144 target = (ShipAPI) ship.getAIFlags().getCustom(AIFlags.TARGET_FOR_SHIP_SYSTEM);
145 }
146
147 if (target != null) {
148 float dist = Misc.getDistance(ship.getLocation(), target.getLocation());
149 float radSum = ship.getCollisionRadius() + target.getCollisionRadius();
150 if (dist > range + radSum) target = null;
151 } else {
152 if (target == null || target.getOwner() == ship.getOwner()) {
153 if (player) {
154 target = Misc.findClosestShipEnemyOf(ship, ship.getMouseTarget(), HullSize.FIGHTER, range, true);
155 } else {
156 Object test = ship.getAIFlags().getCustom(AIFlags.MANEUVER_TARGET);
157 if (test instanceof ShipAPI) {
158 target = (ShipAPI) test;
159 float dist = Misc.getDistance(ship.getLocation(), target.getLocation());
160 float radSum = ship.getCollisionRadius() + target.getCollisionRadius();
161 if (dist > range + radSum) target = null;
162 }
163 }
164 }
165 if (target == null) {
166 target = Misc.findClosestShipEnemyOf(ship, ship.getLocation(), HullSize.FIGHTER, range, true);
167 }
168 }
169
170 return target;
