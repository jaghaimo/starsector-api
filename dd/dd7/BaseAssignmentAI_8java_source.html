372// if (fleet.getFaction().getId().equals(Factions.DIKTAT)) {

376 for (MarketAPI market : Misc.getMarketsInLocation(fleet.getContainingLocation())) {

378 if (!delegate.canRaid(fleet, market)) continue;
379 } else {

381 }
382
383 float dist = Misc.getDistance(fleet, market.getPrimaryEntity());
384 if (dist < minDist) {
385 closest = market;
386 minDist = dist;

388 }
389
390 if (closest == null || minDist > 2000f) return;
391
392 for (CampaignFleetAPI other : Misc.getNearbyFleets(closest.getPrimaryEntity(), 2000f)) {
393 if (other == fleet) continue;
394
395 if (other.isHostileTo(fleet)) {
396 VisibilityLevel vis = other.getVisibilityLevelTo(fleet);
397 boolean canSee = vis == VisibilityLevel.COMPOSITION_AND_FACTION_DETAILS || vis == VisibilityLevel.COMPOSITION_DETAILS;
398 if (!canSee && other.getFaction() != fleet.getFaction()) continue;
399
400 if (other.getAI() instanceof ModularFleetAIAPI) {
401 ModularFleetAIAPI ai = (ModularFleetAIAPI) other.getAI();
402 if (ai.isFleeing()) continue;
403 if (ai.isMaintainingContact()) continue;
404
405 if (ai.getTacticalModule().getTarget() == fleet) return;
406
407 MemoryAPI mem = other.getMemoryWithoutUpdate();
408 boolean smuggler = mem.getBoolean(MemFlags.MEMORY_KEY_SMUGGLER);
409 boolean trader = mem.getBoolean(MemFlags.MEMORY_KEY_TRADE_FLEET);
410 if (smuggler || trader) continue;
411 }
412 if (other.getFleetPoints() > fleet.getFleetPoints() * 0.25f || other.isStationMode()) return;
413 }
414
415 if (other.getFaction() == fleet.getFaction()) {
416 if (other.getFleetPoints() > fleet.getFleetPoints()) return;
417 if (other.getFleetPoints() == fleet.getFleetPoints()) {
418 float dist = Misc.getDistance(other, closest.getPrimaryEntity());
419 if (dist < minDist) return;
420 }
421 }
422 }
423
424 giveRaidOrder(closest);
