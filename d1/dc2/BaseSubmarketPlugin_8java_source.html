211 if (stack.isCommodityStack()) {

217 com.addTradeMod("buy_" + Misc.genUID(), -qty, TRADE_IMPACT_DAYS);
218 } else if (mode == PlayerEconomyImpactMode.PLAYER_SELL_ONLY || mode == PlayerEconomyImpactMode.NONE) {
