179 public static String pickRole(float quality, FactionAPI faction, PersonImportance imp, Random random) {
180 WeightedRandomPicker<String> picker = new WeightedRandomPicker<String>(random);
181
182 float cycles = PirateBaseManager.getInstance().getDaysSinceStart() / 365f;
183
184 if (imp == PersonImportance.VERY_HIGH && cycles < 3) imp = PersonImportance.HIGH;
185 if (imp == PersonImportance.HIGH && cycles < 1) imp = PersonImportance.MEDIUM;
186
187 if (quality < 0.5f && imp.ordinal() > PersonImportance.MEDIUM.ordinal()) {
188 imp = PersonImportance.MEDIUM;
189 }
190
191 float w = faction.getDoctrine().getWarships() - 1f;
192 float c = faction.getDoctrine().getCarriers() - 1f;
193 float p = faction.getDoctrine().getPhaseShips() - 1f;
194 if (w + c + p < 1) w = 1;
195
196 switch (imp) {
197 case VERY_LOW:
198 picker.add(ShipRoles.COMBAT_SMALL, w);
199 picker.add(ShipRoles.COMBAT_MEDIUM, w/2f);
200 picker.add(ShipRoles.CARRIER_SMALL, c);
201 picker.add(ShipRoles.PHASE_SMALL, p);
202 break;
203 case LOW:
204 picker.add(ShipRoles.COMBAT_SMALL, w/2f);
205 picker.add(ShipRoles.COMBAT_MEDIUM, w);
206 picker.add(ShipRoles.CARRIER_SMALL, c);
207 picker.add(ShipRoles.PHASE_SMALL, p);
