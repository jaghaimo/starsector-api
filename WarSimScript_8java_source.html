484 if (route.getActiveFleet() != null && seenFleets.contains(route.getActiveFleet())) continue;

570 public static LocationDanger getDangerFor(float factionStrength, float enemyStrength) {
571 if (enemyStrength < 100) return LocationDanger.NONE;
572
573 float f = enemyStrength / Math.max(1f, factionStrength + enemyStrength);
574 for (LocationDanger level : LocationDanger.vals) {
575 float test = level.enemyStrengthFraction + (level.next().enemyStrengthFraction - level.enemyStrengthFraction) * 0.5f;
576 if (level == LocationDanger.NONE) test = LocationDanger.NONE.enemyStrengthFraction;
577 if (test >= f) {
578 return level;
579 }
580 }
581 return LocationDanger.EXTREME;
582 }
