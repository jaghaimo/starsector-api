470 fleets.add(other);

472 hostile |= other.getAI().isHostileTo(Global.getSector().getPlayerFleet());

474 hostile |= other.getFaction().isHostileTo(Factions.PLAYER);

477 hostile = true;
478 }

482 if (dist < minDist) minDist = dist;
483 }
484 }
