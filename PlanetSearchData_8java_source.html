192 public List<String> conditions = new ArrayList<>();
193 public ResourceDepositsData(String id, String title) {
194 this.id = id;
195 this.idAny = id + "_any";
196 this.idNone = id + "_none";
197 this.title = title;
198 }
199
200 @Override
201 public boolean accept(SectorEntityToken entity, Map<String, String> params) {
202 String selected = params.get(id);
203 if (selected == null) return true; // invalid UI state
204
205 if (idAny.equals(selected)) {
206 return true;
207 }
208 if (entity.getMarket() == null) {
209 return false;
210 }
211
212 MarketAPI market = entity.getMarket();
213 if (market.getSurveyLevel() != SurveyLevel.FULL) {
214 return false;
215 }
216
217
218 if (idNone.equals(selected)) {
219 for (String cid : conditions) {
220 if (market.hasCondition(cid)) {
221 return false;
222 }
223 }
224 return true;
225 } else {
226 boolean foundAtLeastSelected = false;
227 for (String cid : conditions) {
228 if (cid.equals(selected)) {
229 foundAtLeastSelected = true;
230 }
231 if (foundAtLeastSelected && market.hasCondition(cid)) {
232 return true;
233 }
234 }
235 return false;
236 }
237 }

240 public boolean shouldShow() {
241 return true;
242 }

245 public void createTooltip(TooltipMakerAPI info, float width, String param) {

248 info.addTitle(title + " not required");
249 info.addPara("Disregard the presence or absence of " + title.toLowerCase() + " on the planet, both are acceptable.", opad);
250 } else if (param == idNone) {
251 info.addTitle("No " + title.toLowerCase());
252 info.addPara("Require that no " + title.toLowerCase() + " be present on the planet.", opad);
253 } else if (conditions.contains(param)) {
