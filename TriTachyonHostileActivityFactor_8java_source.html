447 public boolean startMercenaryAttack(MarketAPI source, StarSystemAPI target,
448 EventStageData stage, HostileActivityEventIntel intel, Random random) {
449 if (isPlayerCounterRaidedTriTach()) return false;
450 if (source == null || target == null) return false;
451
452 GenericRaidParams params = new GenericRaidParams(new Random(random.nextLong()), true);
453 params.makeFleetsHostile = false; // will be made hostile when they arrive, not before
454 params.source = source;
455
456 params.prepDays = 7f + random.nextFloat() * 7f;
457 params.payloadDays = 27f + 7f * random.nextFloat();
458
459 params.raidParams.where = target;
460 params.raidParams.type = FGRaidType.SEQUENTIAL;
461
462 Set<String> disrupt = new LinkedHashSet<String>();
463 for (MarketAPI market : Misc.getMarketsInLocation(target, Factions.PLAYER)) {
464 params.raidParams.allowedTargets.add(market);
465 params.raidParams.allowNonHostileTargets = true;
466 for (Industry ind : market.getIndustries()) {
467 if (ind.getSpec().hasTag(Industries.TAG_UNRAIDABLE)) continue;
468 disrupt.add(ind.getId());
469
470 }
471 }
472
473 params.raidParams.disrupt.addAll(disrupt);
474 params.raidParams.raidsPerColony = Math.min(disrupt.size(), 4);
475 if (disrupt.isEmpty()) {
476 params.raidParams.raidsPerColony = 2;
477 }
478
479 if (params.raidParams.allowedTargets.isEmpty()) {
480 return false;
481 }
482
483 params.factionId = Factions.INDEPENDENT;
484 params.style = FleetStyle.QUALITY;
485 params.repImpact = ComplicationRepImpact.NONE;
486
487
488 float fleetSizeMult = 1f;
489
490 float f = intel.getMarketPresenceFactor(target);
491
492 float totalDifficulty = fleetSizeMult * 60f * (0.5f + 0.5f * f);
493
494 totalDifficulty -= 10;
495 params.fleetSizes.add(10); // first size 10 pick becomes the Operational Command
496
497 while (totalDifficulty > 0) {
498 int min = 3;
499 int max = 8;
500
501 //int diff = Math.round(StarSystemGenerator.getNormalRandom(random, min, max));
502 int diff = min + random.nextInt(max - min + 1);
503
504 params.fleetSizes.add(diff);
505 totalDifficulty -= diff;
506 }
507
508 TTMercenaryAttack attack = new TTMercenaryAttack(params);
509 attack.setListener(this);
510 //attack.setPreFleetDeploymentDelay(30f + random.nextFloat() * 60f);
511 //attack.setPreFleetDeploymentDelay(1f);
512 Global.getSector().getIntelManager().addIntel(attack, false);
513
514 return true;
515 }

540
541 for (FleetMemberAPI loss : Misc.getSnapshotMembersLost(otherFleet)) {
542 int fp = loss.getFleetPointCost() * mult;

546 raiderFP += fp;

555 TriTachyonCommerceRaiding.addFactorCreateIfNecessary(factor, null);
556 }
557 }
558 if (raiderFP > 0) {
