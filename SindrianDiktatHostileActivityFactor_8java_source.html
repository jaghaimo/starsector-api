305 List<CompetitorData> data = SindrianDiktatStandardActivityCause.computePlayerCompetitionData();
306 CountingMap<MarketAPI> counts = new CountingMap<MarketAPI>();
307
308 for (CompetitorData curr : data) {
309 for (MarketAPI market : curr.competitorProducers) {

312 int weight = market.getCommodityData(curr.commodityId).getMaxSupply();

363 public boolean startAttack(MarketAPI source, MarketAPI target, StarSystemAPI system, EventStageData stage, Random random) {
364 GenericRaidParams params = new GenericRaidParams(new Random(random.nextLong()), true);
365
366 params.makeFleetsHostile = false; // will be made hostile when they arrive, not before
367
368 params.factionId = source.getFactionId();
369 params.source = source;
370
371 params.prepDays = 14f + random.nextFloat() * 14f;
372 params.payloadDays = 27f + 7f * random.nextFloat();
373
374 params.raidParams.where = system;
375 params.raidParams.type = FGRaidType.SEQUENTIAL;
376 params.raidParams.tryToCaptureObjectives = false;
377 params.raidParams.allowedTargets.add(target);
378 params.raidParams.allowNonHostileTargets = true;
379 params.raidParams.setBombardment(BombardType.SATURATION);
380
381 params.style = FleetStyle.STANDARD;
382
383
384 // standard Askonia fleet size multiplier with no shortages/issues is a bit over 230%
385 float fleetSizeMult = source.getStats().getDynamic().getMod(Stats.COMBAT_FLEET_SIZE_MULT).computeEffective(0f);
386
387 float f = intel.getMarketPresenceFactor(system);
388
389 float totalDifficulty = fleetSizeMult * 15f * (0.6f + 0.4f * f);
390
391 if (totalDifficulty < 15) {
392 return false;
393 }
394 if (totalDifficulty > 100) {
395 totalDifficulty = 100;
396 }
397
398 totalDifficulty -= 10;
399
400 params.fleetSizes.add(10);
401
402 while (totalDifficulty > 0) {
403 int min = 6;
404 int max = 10;
405
406 //int diff = Math.round(StarSystemGenerator.getNormalRandom(random, min, max));
407 int diff = min + random.nextInt(max - min + 1);
408
409 params.fleetSizes.add(diff);
410 totalDifficulty -= diff;
411 }
412
413
414 SindrianDiktatPunitiveExpedition punex = new SindrianDiktatPunitiveExpedition(params);
415 punex.setListener(this);
416 Global.getSector().getIntelManager().addIntel(punex);
417
418// GenericRaidFGI raid = new GenericRaidFGI(params);
419// raid.setListener(this);
420// Global.getSector().getIntelManager().addIntel(raid);
421
422 return true;
423 }
