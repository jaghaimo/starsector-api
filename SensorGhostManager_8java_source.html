106 public void advance(float amount) {
107 if (amount == 0) return;
108 CampaignFleetAPI pf = Global.getSector().getPlayerFleet();
109 if (pf == null) return;
110
111 float days = Global.getSector().getClock().convertToDays(amount);
112 perCreatorTimeouts.advance(days);
113
114 sbTimeoutRemaining -= days;
115 if (sbTimeoutRemaining <= 0f) {
116 sbTimeoutRemaining = 0f;
117 checkSensorBursts();
118 }
119
120 timeoutRemaining -= days * GHOST_SPAWN_RATE_MULT;
121 if (timeoutRemaining <= 0f) {
122 spawnGhost();
123 spawnTriggeredBySensorBurst = false;
124 }
125
126 Iterator<SensorGhost> iter = ghosts.iterator();
127 while (iter.hasNext()) {
128 SensorGhost curr = iter.next();
129 curr.advance(amount);
130 if (curr.isDone()) {
131 iter.remove();
132 }
133 }

140 public void checkSensorBursts() {
141 if (!Global.getSector().getCurrentLocation().isHyperspace()) return;
142 if (timeoutRemaining < 1f) return;
143 if (Global.getSector().getMemoryWithoutUpdate().getBoolean(MemFlags.GLOBAL_SENSOR_BURST_JUST_USED_IN_CURRENT_LOCATION)) {
144 if (random.nextFloat() > SB_ATTRACT_GHOSTS_PROBABILITY) {
145 sbTimeoutRemaining = MIN_SB_TIMEOUT + (MAX_SB_TIMEOUT - MIN_SB_TIMEOUT) * random.nextFloat();
146 sbTimeoutRemaining *= SB_FAILED_TO_ATTRACT_TIMEOUT_MULT;
147 return;
148 }
149 CampaignFleetAPI pf = Global.getSector().getPlayerFleet();
150 float range = 2000f;
151 for (CampaignFleetAPI fleet : Global.getSector().getCurrentLocation().getFleets()) {
152 float dist = Misc.getDistance(fleet.getLocation(), pf.getLocation());
153 if (dist > range) continue;
154 if (fleet.getMemoryWithoutUpdate().getBoolean(MemFlags.JUST_DID_SENSOR_BURST)) {
155 timeoutRemaining = 0.2f + 0.8f * random.nextFloat();
156 spawnTriggeredBySensorBurst = true;
157 sbTimeoutRemaining = MIN_SB_TIMEOUT + (MAX_SB_TIMEOUT - MIN_SB_TIMEOUT) * random.nextFloat();
158 break;
159 }
160 }
161 }

164 public void spawnGhost() {
165 CampaignFleetAPI pf = Global.getSector().getPlayerFleet();
166 boolean nearStream = Misc.isInsideSlipstream(pf.getLocation(), 1000f, pf.getContainingLocation());
167
168 WeightedRandomPicker<SensorGhostCreator> picker = new WeightedRandomPicker<SensorGhostCreator>(random);
169 for (SensorGhostCreator creator : CREATORS) {
170 if (perCreatorTimeouts.contains(creator.getId())) continue;
171 if (nearStream && !creator.canSpawnWhilePlayerInOrNearSlipstream()) continue;
172
173 float freq = creator.getFrequency(this);
174 picker.add(creator, freq);
175 }
176
177 SensorGhostCreator creator = picker.pick();
178 if (creator == null) return;
179
180 //System.out.println("Picked: " + creator.getId());
181
182 boolean canSpawn = true;
183 // important: the creator that can't spawn a ghost can still be picked, just won't fire
184 // otherwise moving in/out of slipstreams would manipulate ghost spawning
185 // can still manipulate it since it causes a "failed to create" timeout rather than a "created" one,
186 // but that should be a bit less noticeable
187// if (!creator.canSpawnWhilePlayerInOrNearSlipstream()) {
188// //CampaignFleetAPI pf = Global.getSector().getPlayerFleet();
189// canSpawn = !Misc.isInsideSlipstream(pf.getLocation(), 1000f, pf.getContainingLocation());
190// }
191
192 List<SensorGhost> ghosts = null;
193 if (canSpawn) {
194 ghosts = creator.createGhost(this);
195 }
196 boolean anyFailed = false; // bit of a failsafe if a creator returns a failed-to-spawn ghost
197 if (ghosts != null) {
198 for (SensorGhost curr : ghosts) {
199 anyFailed |= curr.isCreationFailed();
200 }
201 }
202 if (!canSpawn) {
203 anyFailed = true;
204 }
205
206 if (ghosts == null || ghosts.isEmpty() || anyFailed) {
207 float timeout = MIN_FAILED_CREATOR_TIMEOUT_DAYS +
208 random.nextFloat() * (MAX_FAILED_CREATOR_TIMEOUT_DAYS - MIN_FAILED_CREATOR_TIMEOUT_DAYS);
209 perCreatorTimeouts.set(creator.getId(), timeout);
210 } else {
211 this.ghosts.addAll(ghosts);
212 if (random.nextFloat() < FULL_TIMEOUT_TRIGGER_PROBABILITY) {
213 timeoutRemaining = MIN_FULL_GHOST_TIMEOUT_DAYS +
214 random.nextFloat() * (MAX_FULL_GHOST_TIMEOUT_DAYS - MIN_FULL_GHOST_TIMEOUT_DAYS);
215 } else {
216 timeoutRemaining = MIN_SHORT_GHOST_TIMEOUT_DAYS +
217 random.nextFloat() * (MAX_SHORT_GHOST_TIMEOUT_DAYS - MIN_SHORT_GHOST_TIMEOUT_DAYS);
218 }
219 perCreatorTimeouts.set(creator.getId(), creator.getTimeoutDaysOnSuccessfulCreate(this));
220 }
