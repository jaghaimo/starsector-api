257 public MarketAPI findRaidSource(HostileActivityEventIntel intel, EventStageData stage, final StarSystemAPI target) {
258 if (target == null) return null;
259
260 List<MarketAPI> list = new ArrayList<MarketAPI>();
261 float maxDist = Global.getSettings().getFloat("sectorWidth") * 0.5f;
262
263 for (IntelInfoPlugin curr : Global.getSector().getIntelManager().getIntel(PirateBaseIntel.class)) {
264 PirateBaseIntel base = (PirateBaseIntel) curr;
265 if (base.playerHasDealWithBaseCommander()) continue;
266
267 float dist = Misc.getDistance(target.getLocation(), base.getMarket().getLocationInHyperspace());
268 if (dist > maxDist) continue;
269
270 list.add(base.getMarket());
271 }
272
273 for (MarketAPI market : Global.getSector().getEconomy().getMarketsCopy()) {
274 if (Factions.PIRATES.equals(market.getFaction().getId())) {
275 for (MarketAPI other : Misc.getMarketsInLocation(market.getContainingLocation())) {
276 if (other == market) continue;
277 if (!other.getFaction().isHostileTo(market.getFaction())) continue;
278 if (other.getSize() <= market.getSize() - 2) continue;
279
280 float dist = Misc.getDistance(market.getPrimaryEntity().getLocation(), other.getPrimaryEntity().getLocation());
281 if (dist < 8000) continue;
282
283 list.add(market);
284 }
285 }
286 }
287
288 Collections.sort(list, new Comparator<MarketAPI>() {
289 public int compare(MarketAPI m1, MarketAPI m2) {
290 float d1 = Misc.getDistance(target.getLocation(), m1.getLocationInHyperspace());
291 float d2 = Misc.getDistance(target.getLocation(), m2.getLocationInHyperspace());
292 return (int) Math.signum(d1 - d2);
293 }
294 });
295
296 WeightedRandomPicker<MarketAPI> picker = new WeightedRandomPicker<MarketAPI>(intel.getRandom());
297 for (int i = 0; i < list.size() && i < 4; i++) {
298 MarketAPI market = list.get(i);
299 float dist = Misc.getDistance(target.getLocation(), market.getLocationInHyperspace());
300 float w = 100000f / (dist * dist);
301 picker.add(market, w);
302 }
303
304 return picker.pick();
305 }

309 public boolean startRaid(MarketAPI source, StarSystemAPI target, float raidFP) {
310 boolean hasTargets = false;
311 for (MarketAPI curr : Misc.getMarketsInLocation(target)) {
312 if (curr.getFaction().isHostileTo(source.getFaction())) {
313 hasTargets = true;
314 break;
315 }
316 }
317
318 if (!hasTargets) return false;
319
320 FactionAPI faction = source.getFaction();
321
322 RaidIntel raid = new RaidIntel(target, faction, this);
323
324 float successMult = 0.5f;
325
326 JumpPointAPI gather = null;
327 List<JumpPointAPI> points = source.getContainingLocation().getEntities(JumpPointAPI.class);
328 float min = Float.MAX_VALUE;
329 for (JumpPointAPI curr : points) {
330 float dist = Misc.getDistance(source.getPrimaryEntity().getLocation(), curr.getLocation());
331 if (dist < min) {
332 min = dist;
333 gather = curr;
334 }
335 }
336
337 SectorEntityToken raidJump = RouteLocationCalculator.findJumpPointToUse(faction, target.getCenter());
338 if (gather == null || raidJump == null) return false;
339
340 PirateRaidAssembleStage2 assemble = new PirateRaidAssembleStage2(raid, gather);
341 assemble.addSource(source);
342 assemble.setSpawnFP(raidFP);
343 assemble.setAbortFP(raidFP * successMult);
344 raid.addStage(assemble);
345
346 TravelStage travel = new TravelStage(raid, gather, raidJump, false);
347 travel.setAbortFP(raidFP * successMult);
348 raid.addStage(travel);
349
350 PirateRaidActionStage action = new PirateRaidActionStage(raid, target);
351 action.setAbortFP(raidFP * successMult);
352 raid.addStage(action);
353
354 raid.addStage(new ReturnStage(raid));
355
356 Global.getSector().getIntelManager().addIntel(raid);
357
358 return true;
359 }
