61 public void advance(float amount) {
62 CampaignFleetAPI fleet = Global.getSector().getPlayerFleet();
63 if (fleet == null) return;
64
65 if (Global.getSector().getPlayerStats().getSkillLevel(Skills.HULL_RESTORATION) <= 0) {
66 picked = null;
67 dmod = null;
68 return;
69 }

71 float days = Global.getSector().getClock().convertToDays(amount);
72 float rateMult = 1f / (float) MONTHS_PER_DMOD_REMOVAL;
73 //days *= 100f;
74 tracker.advance(days * rateMult * 0.5f); // * 0.5f since the tracker interval averages 15 days
75 if (tracker.intervalElapsed()) {
76 // pick which ship to remove which d-mod from half a month ahead of time
77 // if it's no longer present in the fleet when it's time to remove the d-mod,
78 // don't remove a d-mod at all
79 if (picked == null || dmod == null) {
80 pickNext();
81 } else {
82 if (fleet.getFleetData().getMembersListCopy().contains(picked) &&
83 DModManager.getNumDMods(picked.getVariant()) > 0) {
84 DModManager.removeDMod(picked.getVariant(), dmod);
85
86 HullModSpecAPI spec = DModManager.getMod(dmod);
87 MessageIntel intel = new MessageIntel(picked.getShipName() + " - repaired " + spec.getDisplayName(), Misc.getBasePlayerColor());
88 intel.setIcon(Global.getSettings().getSpriteName("intel", "repairs_finished"));
89 Global.getSector().getCampaignUI().addMessage(intel, MessageClickAction.REFIT_TAB, picked);
90
91 int dmods = DModManager.getNumDMods(picked.getVariant());
92 if (dmods <= 0) {
93 restoreToNonDHull(picked.getVariant());
94 }
95 }
96 picked = null;
97 pickNext();
98 }
99 }
100
101 tracker2.advance(days);
102 if (tracker2.intervalElapsed() && REMOVE_DMOD_FROM_NEW_SHIPS) {
103 if (pickedNew == null || dmodNew == null) {
104 pickNextNew();
105 } else {
106 seen.add(pickedNew.getId());
107
108 float numDmods = DModManager.getNumDMods(pickedNew.getVariant());
109 if (fleet.getFleetData().getMembersListCopy().contains(pickedNew) && numDmods > 0) {
110 float probRemove = MIN_NEW_REMOVE_PROB + numDmods * NEW_REMOVE_PROB_PER_DMOD;
111 if (newRandom.nextFloat() < probRemove) {
112 DModManager.removeDMod(pickedNew.getVariant(), dmodNew);
113
114 HullModSpecAPI spec = DModManager.getMod(dmodNew);
115 MessageIntel intel = new MessageIntel(pickedNew.getShipName() + " - repaired " + spec.getDisplayName(), Misc.getBasePlayerColor());
116 intel.setIcon(Global.getSettings().getSpriteName("intel", "repairs_finished"));
117 Global.getSector().getCampaignUI().addMessage(intel, MessageClickAction.REFIT_TAB, pickedNew);
118
119 int dmods = DModManager.getNumDMods(pickedNew.getVariant());
120 if (dmods <= 0) {
121 restoreToNonDHull(pickedNew.getVariant());
122 }
123 }
124 }

131 public void pickNext() {
132 picked = null;
133 dmod = null;
134
135 CampaignFleetAPI fleet = Global.getSector().getPlayerFleet();
136 WeightedRandomPicker<FleetMemberAPI> picker = new WeightedRandomPicker<FleetMemberAPI>();
137 for (FleetMemberAPI member : fleet.getFleetData().getMembersListCopy()) {
138 if (member.getVariant().isStockVariant()) continue;
139 if (member.isMothballed()) continue;
140 if (member.getHullSpec().hasTag(Tags.HULL_UNRESTORABLE) ||
141 member.getVariant().hasTag(Tags.VARIANT_UNRESTORABLE)) continue;
142 int dmods = DModManager.getNumNonBuiltInDMods(member.getVariant());
143 if (dmods > 0) {
144 picker.add(member, 1);
145 }
146 }
147 picked = picker.pick();
148
149 if (picked != null) {
150 ShipVariantAPI variant = picked.getVariant();
151 WeightedRandomPicker<String> modPicker = new WeightedRandomPicker<String>();
152 for (String id : variant.getHullMods()) {
153 if (DModManager.getMod(id).hasTag(Tags.HULLMOD_DMOD)) {
154 if (variant.getHullSpec().getBuiltInMods().contains(id)) continue;
155 modPicker.add(id);
156 }
157 }
158 dmod = modPicker.pick();
159 if (dmod == null) {
160 picked = null;
