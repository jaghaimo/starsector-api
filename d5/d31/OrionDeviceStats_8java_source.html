51 public void advance(float amount) {
52 List<PusherPlateImpulse> remove = new ArrayList<PusherPlateImpulse>();
53 float totalForce = 0f;
54 for (PusherPlateImpulse curr : impulses) {
55 totalForce += curr.force;
56 curr.elapsed += amount;
57 if (curr.elapsed >= curr.dur) {
58 remove.add(curr);
59 }
60 }
61 impulses.removeAll(remove);
62
63 // assuming k of 1, and a mass of 1
64 float springForce = compression;
65 float netForce = totalForce - springForce;
66
67 vel += netForce * amount;
68 compression += vel * amount;
69
70 if (compression > 1f) {
71 compression = 1f;
72 vel = 0f;
73 }
74 float min = 0f;
75 //min = -0.5f;
76 if (compression < min) {
77 compression = min;
