432 public void updateMarketDetectionRanges() {
433 if (isStageActive(Stage.SLIPSTREAM_DETECTION)) {
434 String id1 = "hypertopology1";
435 String id2 = "hypertopology2";
436 String id3 = "hypertopology3";
437 String id4 = "hypertopology4";
438 for (MarketAPI market : Global.getSector().getEconomy().getMarketsCopy()) {
439 if (market.isHidden()) continue;
440
441 boolean unapplicable = false;
442 Industry spaceport = market.getIndustry(Industries.SPACEPORT);
443 if (spaceport == null) {
444 spaceport = market.getIndustry(Industries.MEGAPORT);
445 }
446 if (spaceport == null || !spaceport.isFunctional()) {
447 unapplicable = true;
448 }
449
450 StatBonus mod = market.getStats().getDynamic().getMod(Stats.SLIPSTREAM_REVEAL_RANGE_LY_MOD);
451 if (!market.isPlayerOwned() || unapplicable) {
452 mod.unmodify(id1);
453 mod.unmodify(id2);
454 mod.unmodify(id3);
455 mod.unmodify(id4);
456 continue;
457 }
458
459 mod.modifyFlat(id1, BASE_DETECTION_RANGE_LY, "Base detection range");
460 mod.modifyFlat(id2, market.getSize(), "Colony size");
461
462 float arraysBonus = gerSensorArrayBonusFor(market, RANGE_WITHIN_WHICH_SENSOR_ARRAYS_HELP_LY);
463
464 mod.modifyFlatAlways(id3, arraysBonus,
465 "Claimed sensor arrays within " + (int) RANGE_WITHIN_WHICH_SENSOR_ARRAYS_HELP_LY +
466 " ly (max: " + (int) MAX_SENSOR_ARRAYS + " arrays)");

471 public float gerSensorArrayBonusFor(MarketAPI market, float range) {
472 int countDomain = 0;
473 int countMakeshift= 0;
474 Vector2f locInHyper = market.getLocationInHyperspace();
475 for (StarSystemAPI system : Global.getSector().getStarSystems()) {
476 float dist = Misc.getDistanceLY(locInHyper, system.getLocation());
477 if (dist > range && Math.round(dist * 10f) <= range * 10f) {
478 dist = range;
479 }
480 if (dist <= range) {
481 for (SectorEntityToken entity : system.getEntitiesWithTag(Tags.SENSOR_ARRAY)) {
482 if (entity.getFaction() != null && entity.getFaction().isPlayerFaction()) {
483 if (entity.hasTag(Tags.MAKESHIFT)) {
484 countMakeshift++;
485 } else {
486 countDomain++;
487 }
488 }
489 }
490 }
491 }
492
493 float bonus = Math.min(countDomain, MAX_SENSOR_ARRAYS) * RANGE_PER_DOMAIN_SENSOR_ARRAY;
494 bonus += Math.min(Math.max(0, countMakeshift - countDomain), MAX_SENSOR_ARRAYS) * RANGE_PER_MAKESHIFT_SENSOR_ARRAY;
495
496 return bonus;
497 }
