814 public void setProgress(int progress) {
815 if (this.progress == progress) return;
816
817 if (progress < 0) progress = 0;
818 if (progress > maxProgress) progress = maxProgress;
819
820 EventStageData prev = getLastActiveStage(true);
821 prevProgressDeltaWasPositive = this.progress < progress;
822
823 //progress += 30;
824 //progress = 40;
825 //progress = 40;
826 //progress = 499;
827
828 this.progress = progress;
829
830
831 if (progress < 0) {
832 progress = 0;
833 }
834 if (progress > getMaxProgress()) {
835 progress = getMaxProgress();
836 }
837
838 // Check to see if randomized events need to be rolled/reset
839 for (EventStageData esd : getStages()) {
840 if (esd.wasEverReached && esd.isOneOffEvent && !esd.isRepeatable) continue;
841
842 if (esd.randomized) {
843 if (esd.rollData != null && progress <= esd.progressToResetAt) {
844 resetRandomizedStage(esd);
845 }
846 if (esd.rollData == null && progress >= esd.progressToRollAt) {
847 rollRandomizedStage(esd);
848 if (esd.rollData == null) {
849 esd.rollData = RANDOM_EVENT_NONE;
850 }
851 }
852 }
853 }
854
855 // go through all of the stages made active by the new progress value
856 // generally this'd just be one stage, but possible to have multiple for a large
857 // progress increase
858 for (EventStageData curr : getStages()) {
859 if (curr.progress <= prev.progress) continue;
860 //if (curr.progress > progress) continue;
861
862 // reached
863 if (curr.progress <= progress) {
864 //EventStageData curr = getLastActiveStage(true);
865 if (curr != null && (curr != prev || !prev.wasEverReached)) {
866 if (curr.sendIntelUpdateOnReaching && curr.progress > 0 && (prev == null || prev.progress < curr.progress)) {
867 sendUpdateIfPlayerHasIntel(curr, getTextPanelForStageChange());
868 }
869 notifyStageReached(curr);
870 curr.rollData = null;
871 curr.wasEverReached = true;
872
873 progress = getProgress(); // in case it was changed by notifyStageReached()
874 }
875 }
