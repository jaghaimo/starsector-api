344 protected MarketAPI getNearestMarket(boolean playerOnly) {
345 MarketAPI nearest = null;
346 float minDist = Float.MAX_VALUE;
347 CampaignFleetAPI pf = Global.getSector().getPlayerFleet();
348 for (MarketAPI curr : Global.getSector().getEconomy().getMarketsCopy()) {
349 if (curr.isHidden()) continue;
350 if (playerOnly && !curr.isPlayerOwned()) continue;
351
352 float dist = Misc.getDistanceLY(pf, curr.getPrimaryEntity());
353 boolean nearer = dist < minDist;
354 if (dist == minDist && dist == 0 && nearest != null) {
355 float d1 = Misc.getDistance(pf, curr.getPrimaryEntity());
356 float d2 = Misc.getDistance(pf, nearest.getPrimaryEntity());
357 nearer = d1 < d2;
358 }
359 if (nearer) {
360 nearest = curr;
361 minDist = dist;
362 }
363 }
364 return nearest;
365 }
