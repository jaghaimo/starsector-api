66 protected void updateDPFromSupportDoctrine() {
67 // making sure that while transferring command away from something, the previous player ship
68 // does NOT receive the DP reduction until transferring command is finished
69
70 ShipAPI from = engine.getShipPlayerIsTransferringCommandFrom();
71 String id = SupportDoctrine.SUPPORT_DOCTRINE_DP_REDUCTION_ID + "_reverse";
72 if (from != null) {
73 StatMod bonus = from.getMutableStats().getDynamic().getMod(Stats.DEPLOYMENT_POINTS_MOD).getFlatBonus(SupportDoctrine.SUPPORT_DOCTRINE_DP_REDUCTION_ID);
74 if (bonus != null && bonus.value != 0) {
75 undoDPMod = from;
76 from.getMutableStats().getDynamic().getMod(Stats.DEPLOYMENT_POINTS_MOD).modifyFlat(id, -bonus.value);
77 if (from.getFleetMember() != null) {
78 from.getFleetMember().getStats().getDynamic().getMod(Stats.DEPLOYMENT_POINTS_MOD).modifyFlat(id, -bonus.value);
79 }
80 }
81 } else if (undoDPMod != null) {
82 undoDPMod.getMutableStats().getDynamic().getMod(Stats.DEPLOYMENT_POINTS_MOD).unmodifyFlat(id);

84 undoDPMod.getFleetMember().getStats().getDynamic().getMod(Stats.DEPLOYMENT_POINTS_MOD).unmodifyFlat(id);
