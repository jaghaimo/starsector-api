195 public NascentGravityWellAPI getNearestWell(float maxDist) {
196 CampaignFleetAPI fleet = getFleet();
197 if (fleet == null) return null;
198 if (!fleet.isInHyperspace()) return null;
199
200 float minDist = Float.MAX_VALUE;
201 NascentGravityWellAPI closest = null;
202 List<Object> wells = fleet.getContainingLocation().getEntities(NascentGravityWellAPI.class);
203 for (Object o : wells) {
204 NascentGravityWellAPI well = (NascentGravityWellAPI) o;
205 float dist = Misc.getDistance(well.getLocation(), fleet.getLocation());
206 dist -= well.getRadius() + fleet.getRadius();
207 if (dist > maxDist) continue;
208 if (dist < minDist) {
209 minDist = dist;
210 closest = well;
211 }
212 }
213 return closest;
214 }
