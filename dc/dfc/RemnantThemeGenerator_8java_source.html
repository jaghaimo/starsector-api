459 public static CustomCampaignEntityAPI addBeacon(StarSystemAPI system, RemnantSystemType type) {
460
461 SectorEntityToken anchor = system.getHyperspaceAnchor();
462 List<SectorEntityToken> points = Global.getSector().getHyperspace().getEntities(JumpPointAPI.class);
463
464 float minRange = 600;
465
466 float closestRange = Float.MAX_VALUE;
467 JumpPointAPI closestPoint = null;
468 for (SectorEntityToken entity : points) {
469 JumpPointAPI point = (JumpPointAPI) entity;
470
471 if (point.getDestinations().isEmpty()) continue;

473 JumpDestination dest = point.getDestinations().get(0);

572 });
