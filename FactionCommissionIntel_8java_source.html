158 public void undoRepChange(FactionAPI other, InteractionDialogAPI dialog) {
159 String id = other.getId();
160 RepChangeData change = repChanges.get(id);
161
162 if (change == null) return;
163
164 CustomRepImpact impact = new CustomRepImpact();
165 impact.delta = -change.delta;
166 impact.delta = Math.max(0f, impact.delta - Global.getSettings().getFloat("factionCommissionRestoredRelationshipPenalty"));
167 if (impact.delta > 0) {
168 Global.getSector().adjustPlayerReputation(
169 new RepActionEnvelope(RepActions.CUSTOM,
170 impact, null, dialog != null ? dialog.getTextPanel() : null, false, true),
171 id);
172 }
173 repChanges.remove(id);
174 }

219 public void reportBattleOccurred(CampaignFleetAPI fleet, CampaignFleetAPI primaryWinner, BattleAPI battle) {
220 if (isEnded() || isEnding()) return;
221
222 if (!battle.isPlayerInvolved()) return;
223
224 int payment = 0;
225 float fpDestroyed = 0;
226 for (CampaignFleetAPI otherFleet : battle.getNonPlayerSideSnapshot()) {
227 if (!faction.isHostileTo(otherFleet.getFaction())) continue;
228
229 float bounty = 0;
230 for (FleetMemberAPI loss : Misc.getSnapshotMembersLost(otherFleet)) {
231 float mult = Misc.getSizeNum(loss.getHullSpec().getHullSize());
232 bounty += mult * baseBounty;
233 fpDestroyed += loss.getFleetPointCost();
234 }
235
236 payment += (int) (bounty * battle.getPlayerInvolvementFraction());
237 }
238
239 if (payment > 0) {
240 Global.getSector().getPlayerFleet().getCargo().getCredits().add(payment);
241
242 float repFP = (int)(fpDestroyed * battle.getPlayerInvolvementFraction());
243 ReputationAdjustmentResult rep = Global.getSector().adjustPlayerReputation(
244 new RepActionEnvelope(RepActions.COMMISSION_BOUNTY_REWARD, new Float(repFP), null, null, true, false),
245 faction.getId());
246 latestResult = new CommissionBountyResult(payment, battle.getPlayerInvolvementFraction(), rep);
247 sendUpdateIfPlayerHasIntel(latestResult, false);
248 }
