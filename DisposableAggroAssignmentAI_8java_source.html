113 public static SectorEntityToken pickEntityToGuard(Random random, StarSystemAPI system, CampaignFleetAPI fleet) {
114 WeightedRandomPicker<SectorEntityToken> picker = new WeightedRandomPicker<SectorEntityToken>(random);
115
116 for (SectorEntityToken entity : system.getJumpPoints()) {
117 float mult = getWeightMultForPatrols(system, fleet, entity);
118 picker.add(entity, 5f * mult);
119 }
120
121 // gas giants are exits from hyperspace, so hang around there too
122 for (PlanetAPI planet : system.getPlanets()) {
123 if (planet.isGasGiant()) {
124 float mult = getWeightMultForPatrols(system, fleet, planet);
125 picker.add(planet, 5f * mult);
126 }
127 }
128
129 for (MarketAPI market : Global.getSector().getEconomy().getMarkets(system)) {
130 if (market.getFaction() != fleet.getFaction()) continue;
131
132 float mult = getWeightMultForPatrols(system, fleet, market.getPrimaryEntity());
133 picker.add(market.getPrimaryEntity(), 5f * mult);
134 }
135
136 return picker.pick();
137 }

144 public static float countNearbyPatrols(StarSystemAPI system, CampaignFleetAPI fleet, SectorEntityToken entity) {
145 float count = 0;
146 for (CampaignFleetAPI other : system.getFleets()) {
147 float dist = Misc.getDistance(other, entity);
148 if (dist > 3000) continue;
149
150 if (!Misc.isPatrol(other)) continue;
151

154 }
