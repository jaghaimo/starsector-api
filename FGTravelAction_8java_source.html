63 public static float computeETADays(CampaignFleetAPI fleet, SectorEntityToken dest) {
64 boolean sameLoc = fleet.getContainingLocation() == dest.getContainingLocation();
65 float dist = 0f;
66 boolean destIsSystem = dest.getStarSystem() != null && dest.getStarSystem().getCenter() == dest;
67 if (sameLoc) {
68 dist = Misc.getDistance(fleet, dest);
69 if (destIsSystem) dist = 0f;
70 } else {
71 dist = Misc.getDistance(fleet.getLocationInHyperspace(), dest.getLocationInHyperspace());
72
73 if (!destIsSystem) {
74 dist += 5000f; // fudge factor for traveling in-system
75 }
76
77 if (fleet.getContainingLocation() != null &&
78 dest.getContainingLocation() != null &&
79 !fleet.getContainingLocation().isHyperspace() &&
80 !dest.getContainingLocation().isHyperspace()) {
81 dist += 5000f; // two legs of travel are in-system
82 }
83 }
84
85 return dist / 1000f;
