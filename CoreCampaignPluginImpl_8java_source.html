381 public static final float computeSmugglingSuspicionLevel(MarketAPI market) {
382 float smugglingTotal = 0f;
383 float tradeTotal = 0f;
384 for (SubmarketAPI submarket : market.getSubmarketsCopy()) {
385 SubmarketPlugin plugin = submarket.getPlugin();
386 if (!plugin.isParticipatesInEconomy()) continue;
387
388 PlayerTradeDataForSubmarket tradeData = SharedData.getData().getPlayerActivityTracker().getPlayerTradeData(submarket);
389 if (submarket.getFaction().isHostileTo(market.getFaction()) || submarket.getPlugin().isBlackMarket()) {
390 smugglingTotal += tradeData.getTotalPlayerTradeValue();
391 } else {
392 tradeTotal += tradeData.getTotalPlayerTradeValue();
393 }
394 }
395
396 float suspicionLevel = 0f;
397
398 if (smugglingTotal + tradeTotal > 0) {
399 suspicionLevel = smugglingTotal / (smugglingTotal + tradeTotal);
400
401 //float playerImpactMult = Global.getSettings().getFloat("economyPlayerTradeImpactMult");
402 //float threshold = 30000f * playerImpactMult * market.getSize();
403 float threshold = 10000f * market.getSize();
404 suspicionLevel *= Math.min(1f, (smugglingTotal + tradeTotal) / threshold);
405 }
406
407 float extra = market.getMemoryWithoutUpdate().getFloat(MemFlags.MARKET_EXTRA_SUSPICION);
408 suspicionLevel += extra;
409 suspicionLevel = Math.min(1f, suspicionLevel);
410
411 return suspicionLevel;
412 }
