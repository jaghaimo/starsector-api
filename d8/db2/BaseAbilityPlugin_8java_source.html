116 public List<AbilityPlugin> getInterruptedList() {
117 List<AbilityPlugin> result = new ArrayList<AbilityPlugin>();
118 CampaignFleetAPI fleet = getFleet();
119 if (fleet == null) return result;
120
121 for (AbilityPlugin curr : fleet.getAbilities().values()) {
122 if (curr == this) continue;
123
124 if (this instanceof BaseToggleAbility && curr instanceof BaseDurationAbility) {
125 continue;
126 }
127 if (!isCompatible(curr)) {
128 result.add(curr);
129 }
130 }
131 Collections.sort(result, new Comparator<AbilityPlugin>() {
132 public int compare(AbilityPlugin o1, AbilityPlugin o2) {
133 return o1.getSpec().getSortOrder() - o2.getSpec().getSortOrder();
134 }
135 });
136 return result;
137 }
