253 public static float getTurnDir(Vector2f p1, Vector2f p2, Vector2f p3) {
254 float r = (p2.x - p1.x) * (p3.y - p1.y) - (p2.y - p1.y) * (p3.x - p1.x);

256 if (r < 0) return -1f; // right turn
