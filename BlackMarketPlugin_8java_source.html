198 public static void delayedLearnBlueprintsFromTransaction(FactionAPI faction, CargoAPI cargo, PlayerMarketTransaction transaction, float daysDelay) {
199 DelayedBlueprintLearnScript script = new DelayedBlueprintLearnScript(faction.getId(), daysDelay);
200 for (CargoStackAPI stack : transaction.getSold().getStacksCopy()) {
201 SpecialItemPlugin plugin = stack.getPlugin();
202 if (plugin instanceof BlueprintProviderItem) {
203 BlueprintProviderItem bpi = (BlueprintProviderItem) plugin;
204
205 boolean learnedSomething = false;
206 if (bpi.getProvidedFighters() != null) {
207 for (String id : bpi.getProvidedFighters()) {
208 if (faction.knowsFighter(id)) continue;
209 script.getFighters().add(id);
210 learnedSomething = true;
211 }
212 }
213 if (bpi.getProvidedWeapons() != null) {
214 for (String id : bpi.getProvidedWeapons()) {
215 if (faction.knowsWeapon(id)) continue;
216 script.getWeapons().add(id);
217 learnedSomething = true;
218 }
219 }
220 if (bpi.getProvidedShips() != null) {
221 for (String id : bpi.getProvidedShips()) {
222 if (faction.knowsShip(id)) continue;
223 script.getShips().add(id);
224 learnedSomething = true;
225 }
226 }
227 if (bpi.getProvidedIndustries() != null) {
228 for (String id : bpi.getProvidedIndustries()) {
229 if (faction.knowsIndustry(id)) continue;
230 script.getIndustries().add(id);
231 learnedSomething = true;
232 }
233 }
234
235 if (learnedSomething) {
236 cargo.removeItems(stack.getType(), stack.getData(), 1);
237 }
238 }
239 }
240
241 if (!script.getFighters().isEmpty() || !script.getWeapons().isEmpty() ||
242 !script.getShips().isEmpty() || !script.getIndustries().isEmpty()) {
243 Global.getSector().addScript(script);
244 cargo.sort();
245 }
