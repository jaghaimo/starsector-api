51 public Object getStateTokenForCampaignLocation() {
52 CampaignFleetAPI playerFleet = Global.getSector().getPlayerFleet();
53 if (playerFleet.getContainingLocation() instanceof StarSystemAPI) {
54 /*
55 Just returning CAMPAIGN_SYSTEM and letting getMusicSetIdForCampaignStateToken()
56 return the musicSetId misses the case where there's a transition between
57 two star systems with different music, without hyperspace travel in between -
58 since there would be no state transition (due to it being the same CAMPAIGN_SYSTEM state),
59 a restart of the music - and a change to a different music set - would not be triggered.
60
61 Since the state ID includes the music set id, this means that moving between
62 two systems with the same $musicSetId will not trigger a state change and
63 the same music will just keep playing uninterrupted.
64 */
65 StarSystemAPI system = (StarSystemAPI) playerFleet.getContainingLocation();
66 String musicSetId = system.getMemoryWithoutUpdate().getString(MUSIC_SET_MEM_KEY);
67 if (musicSetId != null) {
68 return getToken(SYSTEM_MUSIC_PREFIX + musicSetId);
69 }
70 return CAMPAIGN_SYSTEM;
71 }
72 return CAMPAIGN_HYPERSPACE;
73 }
