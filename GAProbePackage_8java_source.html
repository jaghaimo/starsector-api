87 // PROB_IT_BLEW_UP = 1f;

89

91 variation = Variation.IT_BLEW_UP;

109 if (probe == null) return false;

111 probe.setId("gaProbe_probe");
112 makeImportant(probe, "$gaProbe_probe", Stage.GO_TO_PROBE);

114

138 triggerSetFleetMissionRef("$gaProbe_ref");

140 triggerOrderFleetPatrol(probe);
141 triggerOrderFleetInterceptPlayer();
142 triggerPickLocationAroundEntity(probe, 1000f);
143 triggerSpawnFleetAtPickedLocation("$gaProbe_patherPermanentFlag", null);
144 triggerFleetMakeImportant("$gaProbe_patherMissionInProgress", Stage.GO_TO_PROBE);
145 endTrigger();
146 }
147 }
148
149 if (variation == Variation.SCAVENGER) {
150 beginWithinHyperspaceRangeTrigger(object, 1f, false, Stage.GO_TO_PROBE);
151 triggerCreateFleet(FleetSize.MEDIUM, FleetQuality.LOWER, Factions.SCAVENGERS, FleetTypes.SCAVENGER_MEDIUM, system);
152 triggerAutoAdjustFleetStrengthMajor();
153 triggerSetFleetFaction(Factions.INDEPENDENT);
154
155 triggerMakeLowRepImpact();
156 triggerFleetSetAvoidPlayerSlowly();
157 triggerMakeFleetIgnoredByOtherFleets();
158
159 triggerPickLocationAtClosestToEntityJumpPoint(system, probe);
160 triggerSetEntityToPickedJumpPoint();
161 triggerPickLocationAroundEntity(5000);
162
163 // so the flag is there prior to the FIND_SCAVENGER stage
164 triggerSpawnFleetAtPickedLocation("$gaProbe_scavenger", null);
165
166 triggerFleetSetTravelActionText("exploring system");

168 triggerOrderFleetPatrolEntity(false);
169

171 triggerFleetAddDefeatTrigger("gaProbeScavengerDefeated");

175 setMapMarkerNameColorBasedOnStar(system);
