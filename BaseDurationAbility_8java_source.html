126 float d = getDeactivationDays();
127 float t = getTotalDurationDays();
128 if (activeDaysLeft > t - a) {
129 if (a <= 0) {
130 level = 1;
131 } else {
132 level = 1f - (activeDaysLeft - (t - a)) / a;
133 }
134 } else if (activeDaysLeft < d) {
135 fadingOut = true;
