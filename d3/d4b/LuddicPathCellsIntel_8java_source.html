723 public void doIncident() {
724 if (incidentType == null) return;
725
726 prevIncidentData = null;
727
728 boolean success = checkSuccess();
729
730 if (incidentType == IncidentType.REDUCED_STABILITY) {

737 WeightedRandomPicker<Industry> picker = new WeightedRandomPicker<Industry>(random);
738 for (Industry ind : market.getIndustries()) {
739 if (!ind.canBeDisrupted()) continue;
740 picker.add(ind, ind.getPatherInterest());
741 }
742 Industry target = picker.pick();
743 if (target == null) {
744 abortIncident();
745 return;
746 }

748 prevIncidentData = target;
749 if (success) {
750 float disruptionDur = MIN_SABOTAGE + random.nextFloat() * (MAX_SABOTAGE - MIN_SABOTAGE);
751 target.setDisrupted(disruptionDur, true);
752 }
753 } else if (incidentType == IncidentType.PLANETBUSTER) {
754 // ??? turn into lava planet, remove market, etc

758 sincePrevIncident = 0f;
759 prevIncidentSucceeded = success;

761
762 if (DebugFlags.SEND_UPDATES_WHEN_NO_COMM || Global.getSector().getIntelManager().isPlayerInRangeOfCommRelay()
