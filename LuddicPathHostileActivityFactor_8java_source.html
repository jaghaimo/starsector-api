302 public MarketAPI pickTargetMarket() {
303 WeightedRandomPicker<MarketAPI> picker = new WeightedRandomPicker<MarketAPI>(getRandomizedStageRandom());
304 for (MarketAPI market : Misc.getPlayerMarkets(false)) {
305 // to put a damper on shenanigans with establishing and abandoning a colony
306 // with an Alpha Core admin to bait an attack
307 if (market.getDaysInExistence() < 180f && !Global.getSettings().isDevMode()) continue;
308
309 LuddicPathCellsIntel cells = LuddicPathCellsIntel.getCellsForMarket(market);
310 if (cells == null || cells.isSleeper()) continue;
311
312 float w = LuddicPathBaseManager.getLuddicPathMarketInterest(market);
313 picker.add(market, w);
314 }
315 return picker.pick();

337 public boolean fireEvent(HostileActivityEventIntel intel, EventStageData stage) {
338 //if (true) return false;
339
340 HAPatherCellsEventData data = getPatherCellData(stage);
341 if (data == null || data.market == null || data.cells == null || data.cells.isSleeper()) {
342 return false;
343 }
344
345 WeightedRandomPicker<Industry> picker = new WeightedRandomPicker<Industry>(intel.getRandom());
346 for (Industry ind : data.market.getIndustries()) {
347 if (!ind.canBeDisrupted()) continue;
348 float mult = 1f;
349 if (ind.isStructure()) mult = 0.1f;
350 picker.add(ind, ind.getPatherInterest() * mult);
351 }
352 Industry target = picker.pick();
353 if (target == null) {
354 return false;
355 }
356
357 data.target = target;
358
359 float disruptionDur = LuddicPathCellsIntel.MIN_SABOTAGE +
360 intel.getRandom().nextFloat() * (LuddicPathCellsIntel.MAX_SABOTAGE - LuddicPathCellsIntel.MIN_SABOTAGE);
361 target.setDisrupted(disruptionDur, true);
362
363 intel.sendUpdateIfPlayerHasIntel(stage.rollData, false);
364
365 return true;
366 }
