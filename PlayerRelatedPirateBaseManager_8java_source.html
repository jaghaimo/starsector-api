262 protected StarSystemAPI pickSystemForPirateBase(StarSystemAPI initialTarget) {
263 WeightedRandomPicker<StarSystemAPI> veryFar = new WeightedRandomPicker<StarSystemAPI>(random);
264 WeightedRandomPicker<StarSystemAPI> far = new WeightedRandomPicker<StarSystemAPI>(random);
265 WeightedRandomPicker<StarSystemAPI> picker = new WeightedRandomPicker<StarSystemAPI>(random);
266
267 for (StarSystemAPI system : Global.getSector().getStarSystems()) {
268 if (system.hasPulsar()) continue;
269
270 float days = Global.getSector().getClock().getElapsedDaysSince(system.getLastPlayerVisitTimestamp());
271 if (days < 180f) continue;
272
273 if (system.getCenter().getMemoryWithoutUpdate().contains(PirateBaseManager.RECENTLY_USED_FOR_BASE)) continue;
274
275 float weight = 0f;
276 if (system.hasTag(Tags.THEME_MISC_SKIP)) {
277 weight = 1f;
278 } else if (system.hasTag(Tags.THEME_MISC)) {
279 weight = 3f;
280 } else if (system.hasTag(Tags.THEME_REMNANT_NO_FLEETS)) {
281 weight = 3f;
282 } else if (system.hasTag(Tags.THEME_REMNANT_DESTROYED)) {
283 weight = 3f;
284 } else if (system.hasTag(Tags.THEME_RUINS)) {
285 weight = 5f;
286 } else if (system.hasTag(Tags.THEME_CORE_UNPOPULATED)) {
287 //weight = 1f;
288 weight = 0f;
289 }
290 if (weight <= 0f) continue;
291
292 float usefulStuff = system.getCustomEntitiesWithTag(Tags.OBJECTIVE).size() +
293 system.getCustomEntitiesWithTag(Tags.STABLE_LOCATION).size();
294 if (usefulStuff <= 0) continue;
295
296 if (Misc.getMarketsInLocation(system).size() > 0) continue;
297
298 float dist = Misc.getDistance(initialTarget.getLocation(), system.getLocation());
299
300 float distMult = 100000f / dist;
301 distMult *= distMult;
302
303 if (dist > 30000f) {
304 veryFar.add(system, weight * usefulStuff * distMult);
305 } else if (dist > 10000f) {
306 far.add(system, weight * usefulStuff * distMult);
307 } else {
308 picker.add(system, weight * usefulStuff * distMult);
309 }
310 }
311
312 if (picker.isEmpty()) {
313 picker.addAll(far);
314 }
315 if (picker.isEmpty()) {
316 picker.addAll(veryFar);
317 }
318
319 return picker.pick();
