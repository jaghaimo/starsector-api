277 float maxSize = 0f;

284 //mag = 0.95f;
285 mag *= 0.5f;
286 mag *= maxSize / 6f;
287 if (base != null) {
288 mag *= (float)(base.getTier().ordinal() + 1f) / 5f;
289 }
