242// from/to are same
243// hyper - hyper
244// system - system
245// system - hyper
246// hyper - system
247// system - other_system
248
249// Subcase:
250// one end is system center - spawn "somewhere in system" in that case

264 Vector2f loc = null;

266 conLoc = from.getContainingLocation();

268

270 if (progress > 0.03f) {
271 // at a typical orbiting radius
272 loc = Misc.getPointAtRadius(from.getLocation(), from.getRadius() + 100f + (float) Math.random() * 100f);
273 } else {

398 fleet.getContainingLocation().removeEntity(fleet);
399 conLoc.addEntity(fleet);
400 }

405 public static float getVarianceMult(float p) {
406 float varianceMult = 1f;
407 if (p < 0.1f) {
408 varianceMult = p / 0.1f;
409 } else if (p > 0.9f) {
410 varianceMult = (1f - p) / 0.1f;
411 }

419 public static JumpPointAPI findJumpPointToUse(FactionAPI faction, SectorEntityToken from) {
420 float min = Float.MAX_VALUE;
421 JumpPointAPI result = null;
422 float fringeMax = 0;
423 JumpPointAPI fringe = null;
424
425 LocationAPI location = from.getContainingLocation();
426 List<JumpPointAPI> points = location.getEntities(JumpPointAPI.class);
427
428
429 for (JumpPointAPI curr : points) {
430 float dist = Misc.getDistance(from.getLocation(), curr.getLocation());
431 if (dist < min) {
432 min = dist;
433 result = curr;
434 }
435 dist = Misc.getDistance(new Vector2f(), curr.getLocation());
436 if (dist > fringeMax) {
437 fringe = curr;
438 fringeMax = dist;
439 }
440 }
441
442 if (from.getContainingLocation() instanceof StarSystemAPI) {
443 StarSystemAPI system = (StarSystemAPI) from.getContainingLocation();
444 boolean useFringeOnly = !isInControlOfSystemOrEven(faction, system);
445 if (useFringeOnly && fringe != null) {
446 return fringe;
447 }
448 }
449
450 return result;
451 }

453 public static boolean isInControlOfSystemOrEven(FactionAPI faction, StarSystemAPI system) {
454 List<MarketAPI> markets = Misc.getMarketsInLocation(system);
455 int hostileMax = 0;
456 int ourMax = 0;
457 for (MarketAPI market : markets) {
458 if (market.getFaction().isHostileTo(faction)) {
459 hostileMax = Math.max(hostileMax, market.getSize());
460 } else if (market.getFaction() == faction) {
461 ourMax = Math.max(ourMax, market.getSize());
462 }
463 }
464 boolean inControl = ourMax >= hostileMax;
465 return inControl;
466 }
