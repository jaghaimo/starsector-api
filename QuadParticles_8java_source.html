93 Iterator<QuadParticleData> iter = particles.iterator();

101
102 curr.loc.x += curr.vel.x * amount;
103 curr.loc.y += curr.vel.y * amount;
104 if (curr.loc.y < floor + curr.floorMod) {

107 if (floorFriction > 0) {

110 }
